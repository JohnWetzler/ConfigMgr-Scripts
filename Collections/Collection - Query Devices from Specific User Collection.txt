SMS_R_SYSTEM.ResourceID,SMS_R_SYSTEM.ResourceType,SMS_R_SYSTEM.Name,SMS_R_SYSTEM.SMSUniqueIdentifier,SMS_R_SYSTEM.ResourceDomainORWorkgroup,SMS_R_SYSTEM.Client from SMS_R_System    LEFT JOIN SMS_UserMachineRelationship ON       SMS_UserMachineRelationship.ResourceID = SMS_R_System.ResourceId        WHERE SMS_UserMachineRelationship.UniqueUserName IN    (SELECT SMS_R_User.UniqueUserName      FROM SMS_R_User       INNER JOIN SMS_CM_RES_COLL_ACX00146 ON         SMS_CM_RES_COLL_ACX00146.ResourceID = SMS_R_User.ResourceID)
